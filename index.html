        async function updateStatus(name, status, note) {

            try {

                const response = await fetch(CONFIG.SCRIPT_URL, {

                    method: 'POST',

                    body: JSON.stringify({

                        action: 'updateStatus',

                        name: name,

                        status: status,

                        note: note || '',

                        timestamp: new Date().toISOString()

                    })

                });

                

                const result = await response.json();

                if (result.success) {

                    loadData();

                    return true;

                }

                throw new Error(result.error);

            } catch (error) {

                console.error('Error updating status:', error);

                alert('Failed to update. Please try again.');

                return false;

            }

        async function addItem(type) {

            const input = document.getElementById(type === 'names' ? 'newName' : 'newLocation');

            const value = input.value.trim();

            

            if (!value) return;

            

            const current = type === 'names' ? configData.names : configData.locations;

            if (current.includes(value)) {

                alert('Already exists');

                return;

            }

            

            const newItems = [...current, value];

            const success = await updateConfig(type, newItems);

            

            if (success) {

                input.value = '';

            }

        }

        

        async function removeItem(type, value) {

            const current = type === 'names' ? configData.names : configData.locations;

            const newItems = current.filter(item => item !== value);

            await updateConfig(type, newItems);

        }

        

        function getTimeClass(timestamp) {

            const minutes = getMinutesAgo(timestamp);

            if (minutes > 60) return 'very-stale';

            if (minutes > 15) return 'stale';

            return '';

        }

        

        function getMinutesAgo(timestamp) {

            if (!timestamp) return 999;

            const then = new Date(timestamp);

            const now = new Date();

            return Math.floor((now - then) / 60000);

        }

        

        function formatTime(timestamp) {

            if (!timestamp) return 'Unknown';

            

            if (timeFormatAbsolute) {

                const date = new Date(timestamp);

                return date.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });

            } else {

                const minutes = getMinutesAgo(timestamp);

                if (minutes < 1) return 'Just now';

                if (minutes === 1) return '1 min ago';

                if (minutes < 60) return `${minutes} min ago`;

                const hours = Math.floor(minutes / 60);

                if (hours === 1) return '1 hour ago';

                return `${hours} hours ago`;

            }

        }

        

        function escapeHtml(text) {

            if (!text) return '';

            const div = document.createElement('div');

            div.textContent = text;

            return div.innerHTML;

        }
